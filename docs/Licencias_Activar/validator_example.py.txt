import json
import uuid
from cryptography.hazmat.primitives import hashes
from cryptography.hazmat.primitives.asymmetric import padding
from cryptography.hazmat.primitives import serialization
from cryptography.exceptions import InvalidSignature

def get_hardware_id():
    """
    Genera un ID de hardware único basado en la dirección MAC de la máquina.
    Este método es confiable en la mayoría de las plataformas.
    """
    # uuid.getnode() devuelve la dirección MAC como un entero de 48 bits.
    # Lo formateamos como una cadena hexadecimal estándar.
    mac = uuid.getnode()
    hardware_id = ':'.join(('%012X' % mac)[i:i+2] for i in range(0, 12, 2))
    return hardware_id

def validate_license(license_filepath, public_key_filepath):
    """
    Valida un archivo de licencia utilizando la clave pública y el ID de hardware local.

    Args:
        license_filepath (str): La ruta al archivo license.json.
        public_key_filepath (str): La ruta al archivo public_key.pem.

    Returns:
        tuple[bool, str]: Una tupla con un booleano (True si es válido) y un mensaje de estado.
    """
    try:
        # --- 1. Cargar la Clave Pública ---
        # El archivo public_key.pem DEBE ser distribuido junto a tu software cliente.
        with open(public_key_filepath, "rb") as key_file:
            public_key = serialization.load_pem_public_key(key_file.read())

        # --- 2. Cargar el archivo de Licencia ---
        with open(license_filepath, 'r') as f:
            license_data = json.load(f)
        
        signature_hex = license_data.get('signature')
        license_info = license_data.get('license_info')

        if not signature_hex or not license_info:
            return False, "Licencia corrupta o malformada."

        # --- 3. Verificar la Firma Digital ---
        # La firma se creó a partir de los bytes UTF-8 del JSON de `license_info` ordenado.
        # Debemos recrear exactamente esos mismos bytes para que la verificación funcione.
        message = json.dumps(license_info, sort_keys=True).encode('utf-8')
        signature = bytes.fromhex(signature_hex)

        try:
            public_key.verify(
                signature,
                message,
                padding.PKCS1v15(),
                hashes.SHA256()
            )
        except InvalidSignature:
            return False, "Firma de licencia inválida. El archivo parece haber sido alterado."

        # --- 4. Si la firma es válida, comparar el Hardware ID ---
        local_hw_id = get_hardware_id()
        license_hw_id = license_info.get('hardware_id')

        if local_hw_id != license_hw_id:
            msg = f"La licencia no es para esta máquina. ID de la licencia: {license_hw_id}, ID de esta máquina: {local_hw_id}"
            return False, msg

        # --- 5. Si todo coincide, la licencia es válida ---
        return True, "Licencia válida."

    except FileNotFoundError as e:
        return False, f"Error: No se encontró un archivo necesario: {e.filename}"
    except Exception as e:
        return False, f"Ocurrió un error inesperado durante la validación: {e}"
