'use server';

import { cookies } from 'next/headers';
import { getDb } from '@/lib/db';
import type { SessionUser } from '@/types/user';
import { addLog } from '@/lib/log-store';
import { revalidatePath } from 'next/cache';
import { redirect } from 'next/navigation';

const SESSION_COOKIE_NAME = 'clic_soporte_session';

// In a real app, you would use a secure password hashing library like bcrypt
// For this prototype, we'll use a simple (and insecure) hashing for demonstration
async function hashPassword(password: string): Promise<string> {
    const buffer = await crypto.subtle.digest('SHA-256', new TextEncoder().encode(password));
    return Buffer.from(buffer).toString('hex');
}

export async function login(previousState: any, formData: FormData) {
    const username = formData.get('username') as string;
    const password = formData.get('password') as string;

    if (!username || !password) {
        return { success: false, message: "Usuario y contraseña son requeridos." };
    }

    // Hardcoded Super Admin check
    if (username === 'Clicadmin' && password === 'Clic2017*') {
        const sessionUser: SessionUser = { id: 'superadmin', username: 'Clicadmin', role: 'superadmin' };
        await createSession(sessionUser);
        addLog({type: 'warn', context: 'Auth', message: `Super admin 'Clicadmin' logged in.`});
        redirect('/');
    }

    try {
        const db = getDb();
        const stmt = db.prepare(`
            SELECT u.id, u.username, u.password_hash, r.name as role 
            FROM users u 
            JOIN roles r ON u.roleId = r.id 
            WHERE u.username = ? AND u.is_active = 1
        `);
        const user = stmt.get(username) as { id: string, username: string, password_hash: string, role: string } | undefined;

        if (!user) {
            addLog({type: 'warn', context: 'Auth', message: `Login failed for user '${username}': user not found or inactive.`});
            return { success: false, message: 'Credenciales inválidas.' };
        }

        const passwordHash = await hashPassword(password);
        if (passwordHash !== user.password_hash) {
            addLog({type: 'warn', context: 'Auth', message: `Login failed for user '${username}': incorrect password.`});
            return { success: false, message: 'Credenciales inválidas.' };
        }

        const sessionUser: SessionUser = { id: user.id, username: user.username, role: user.role };
        await createSession(sessionUser);
        addLog({type: 'info', context: 'Auth', message: `User '${username}' logged in successfully.`});
        
    } catch (error) {
        const errorMessage = error instanceof Error ? error.message : 'Unknown error';
        addLog({type: 'error', context: 'Auth', message: `Login system error: ${errorMessage}`});
        return { success: false, message: 'Error del sistema durante el inicio de sesión.' };
    }
    redirect('/');
}

export async function logout() {
    cookies().delete(SESSION_COOKIE_NAME);
    revalidatePath('/', 'layout');
    redirect('/');
}

async function createSession(user: SessionUser) {
    // In a real app, use a secure session management library like next-auth or lucia-auth.
    // For this prototype, we'll store the user object directly in an encrypted cookie.
    // This is NOT secure for production.
    const sessionData = JSON.stringify(user);
    
    // We'll just base64 encode for this prototype. AGAIN, NOT SECURE.
    const encryptedSessionData = Buffer.from(sessionData).toString('base64');

    cookies().set(SESSION_COOKIE_NAME, encryptedSessionData, {
        httpOnly: true,
        secure: process.env.NODE_ENV === 'production',
        maxAge: 60 * 60 * 24, // 1 day
        path: '/',
    });
}

export async function getCurrentUser(): Promise<SessionUser | null> {
    const sessionCookie = cookies().get(SESSION_COOKIE_NAME);
    if (!sessionCookie) {
        return null;
    }

    try {
        // "Decrypt" the session data
        const decryptedSessionData = Buffer.from(sessionCookie.value, 'base64').toString('utf-8');
        const user = JSON.parse(decryptedSessionData) as SessionUser;
        return user;
    } catch (error) {
        console.error("Failed to parse session cookie:", error);
        return null;
    }
}

// In a real app, you would use a secure password hashing library like bcrypt
// For this prototype, we'll use a simple (and insecure) hashing for demonstration
async function hashPassword(password: string): Promise<string> {
    // In a real app, NEVER store passwords in plaintext or use such a simple hash.
    // Use bcrypt or a similar library.
    const buffer = await crypto.subtle.digest('SHA-256', new TextEncoder().encode(password));
    return Buffer.from(buffer).toString('hex');
}

export async function getUserCount(): Promise<number> {
    try {
        const db = getDb();
        const stmt = db.prepare('SELECT COUNT(*) as count FROM users');
        const result = stmt.get() as { count: number };
        return result.count;
    } catch (error) {
        console.error("Failed to get user count:", error);
        return 0; // Return 0 on error
    }
}

export async function createFirstUser(username: string, password: string, companyName: string): Promise<{success: boolean, message: string}> {
    const userCount = await getUserCount();
    if (userCount > 0) {
        return { success: false, message: "El primer usuario ya ha sido creado." };
    }

    try {
        const db = getDb();
        
        // Ensure roles exist before creating the user
        const roles: RoleName[] = ['Administrador', 'Operador de Consola', 'Supervisor', 'Visualizador'];
        const roleStmt = db.prepare("INSERT OR IGNORE INTO roles (id, name) VALUES (?, ?)");
        const roleMap: { [key in RoleName]: string } = {
            'Administrador': 'admin',
            'Operador de Consola': 'console_operator',
            'Supervisor': 'supervisor',
            'Visualizador': 'viewer',
        };
        roles.forEach(role => roleStmt.run(roleMap[role], role));

        const passwordHash = await hashPassword(password);
        const userId = crypto.randomUUID();
        
        const adminRole = db.prepare("SELECT id FROM roles WHERE name = 'Administrador'").get() as {id: string};
        
        if (!adminRole) {
            throw new Error("El rol de Administrador no existe. No se puede crear el primer usuario.");
        }

        const transaction = db.transaction(() => {
            const userStmt = db.prepare(
                "INSERT INTO users (id, username, password_hash, roleId, is_active, name) VALUES (?, ?, ?, ?, ?, ?)"
            );
            userStmt.run(userId, username, passwordHash, adminRole.id, 1, 'Administrador del Sistema');
        });

        transaction();
        
        // After creating the user, set the company name
        await updateSettings({ companyName: companyName });

        addLog({ type: 'info', context: 'Auth', message: `Primer usuario administrador '${username}' creado.` });
        addLog({ type: 'info', context: 'Settings', message: `Nombre de la empresa establecido como '${companyName}'.` });
        
        revalidatePath('/', 'layout');
        return { success: true, message: `Usuario '${username}' creado.` };

    } catch (error) {
        const errorMessage = error instanceof Error ? error.message : "Unknown error";
        addLog({ type: 'error', context: 'Auth', message: `Error al crear el primer usuario: ${errorMessage}` });
        return { success: false, message: "Error al crear el usuario." };
    }
}
